# Use official slim Python base image
FROM python:3.10-slim as base

# Set working directory inside container
WORKDIR /app

# Environment variable for unbuffered logging (good for Railway logs)
ENV PYTHONUNBUFFERED=1

# Install build dependencies (and pip if not present)
RUN apt-get update && apt-get install -y --no-install-recommends \
    gcc \
    build-essential \
    pkg-config \
    libffi-dev \
    libssl-dev \
    && python3 -m ensurepip --upgrade \
    && python3 -m pip install --upgrade pip \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

# Copy only requirements first for better caching
COPY requirements.txt .

# Cache pip installs
RUN --mount=type=cache,target=/root/.cache/pip \
    python3 -m pip install --no-cache-dir -r requirements.txt

# Now copy rest of the app
COPY . .

# Set PYTHONPATH so imports from 'src/' work
ENV PYTHONPATH=/app/src

# Default start command (can be overridden by Railway)
CMD ["python3", "src/main.py", "--mode", "realtime", "--trade"]
